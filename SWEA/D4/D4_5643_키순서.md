import java.io.BufferedReader;
import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class D4_5643_í‚¤ìˆœì„œ {
	private static int N, M, answer;
	private static int[][] adjMatrix;
	
	public static void main(String[] args) throws IOException {
		System.setIn(new FileInputStream("input_D4_5643_í‚¤ìˆœì„œ.txt"));
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		StringTokenizer st;
		
		int T = Integer.parseInt(br.readLine());
		for (int tc = 1; tc <= T; tc++) {
			N = Integer.parseInt(br.readLine());
			M = Integer.parseInt(br.readLine());
			adjMatrix = new int[N + 1][N + 1];
			answer = 0;
			
			for (int i = 0; i < M; i++) {
				st = new StringTokenizer(br.readLine(), " ");
				int from = Integer.parseInt(st.nextToken());
				int to = Integer.parseInt(st.nextToken());
				
				adjMatrix[from][to] = 1;
			}
			
			// í”Œë¡œì´ë“œ-ì›Œì…œ
			// ê²½ìœ ì§€
			for (int k = 1; k <= N; k++) {
				// ì¶œë°œì§€
				for (int i = 1; i <= N; i++) {
					if (i == k || adjMatrix[i][k] == 0) {
						continue;
					}
					// ë„ì°©ì§€
					for (int j = 1; j <= N; j++) {
						if (adjMatrix[i][j] == 1) continue;
						if (adjMatrix[k][j] == 1) {
							adjMatrix[i][j] = 1;
						}
					}
				}
			}
			
			for (int i = 1; i <= N; i++) {
				for (int j = 1; j <= N; j++) {
					adjMatrix[i][0] += adjMatrix[i][j];
					adjMatrix[0][j] += adjMatrix[i][j];
				}
			}
			
			for (int i = 1; i <= N; i++) {
				if (adjMatrix[i][0] + adjMatrix[0][i] == N - 1) {
					++answer;
				}
			}
			
			sb.append("#").append(tc).append(" ").append(answer).append("\n");
		}
		System.out.print(sb);
	}

}
