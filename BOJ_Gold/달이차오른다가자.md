import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class BOJ_1194_ë‹¬ì´ì°¨ì˜¤ë¥¸ë‹¤ê°€ì {

	static class Point {
		private int x;
		private int y;
		private int count;
		private int key;
		
		public Point(int x, int y, int count, int key) {
			this.x = x;
			this.y = y;
			this.count = count;
			this.key = key;
		}

		public int getX() {
			return x;
		}

		public int getY() {
			return y;
		}

		public int getCount() {
			return count;
		}

		public int getKey() {
			return key;
		}
	}
	
	static int[] dx = {-1, 0, 1, 0};
	static int[] dy = {0, 1, 0, -1};
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		
		int N = Integer.parseInt(st.nextToken()); // ì„¸ë¡œ
		int M = Integer.parseInt(st.nextToken()); // ê°€ë¡œ
		char[][] map = new char[N][M];
		Point start = null;
		
		for (int i = 0; i < N; i++) {
			String str = br.readLine();
			for (int j = 0; j < M; j++) {
				map[i][j] = str.charAt(j);
				if (map[i][j] == '0') {
					start = new Point(i, j, 0, 0);
				}
			}
		}
		
		System.out.println(bfs(map, new boolean[N][M][64], start));
		
	}

	public static int bfs(char[][] map, boolean[][][] visited, Point start) {
		Queue<Point> queue = new LinkedList<Point>();
		queue.offer(start);
		visited[start.getX()][start.getY()][0] = true;
		
		Point cur;
		int x, y, count, key;
		while (!queue.isEmpty()) {
			cur = queue.poll();
			x = cur.getX();
			y = cur.getY();
			count = cur.getCount();
			key = cur.getKey();
			
			if (map[x][y] == '1') {
				return count;
			}
			
			for (int d = 0; d < 4; d++) {
				int tempX = x + dx[d];
				int tempY = y + dy[d];
				int tempKey = key;
				
				if (!check(map, tempX, tempY)) {
					continue;
				}
				
				if (map[tempX][tempY] == '#') {
					continue;
				}
				
				if (visited[tempX][tempY][tempKey]) {
					continue;
				}
				
				if (map[tempX][tempY] >= 'a' && map[tempX][tempY] <= 'f') {
					tempKey |= (1 << map[tempX][tempY] - 'a');
				}
				
				if (map[tempX][tempY] >= 'A' && map[tempX][tempY] <= 'F') {
					if ((tempKey & (1 << (map[tempX][tempY] - 'A'))) == 0) {
						continue;
					}
					
				}
				
				visited[tempX][tempY][tempKey] = true;
				queue.offer(new Point(tempX, tempY, count + 1, tempKey));
			}
		}
		
		return -1;
	}
	
	public static boolean check(char[][] map, int x, int y) {
		return x >= 0 && x < map.length && y >= 0 && y < map[x].length;
	}
}
