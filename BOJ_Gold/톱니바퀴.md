import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class BOJ_14891_í†±ë‹ˆë°”í€´ {

	private static int[] dirs;
	private static int[][] magnets;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringBuilder sb = new StringBuilder();
		StringTokenizer st;

		dirs = new int[4];
		magnets = new int[4][8];

		for (int i = 0; i < 4; i++) {
			String input = br.readLine();
			for (int j = 0; j < 8; j++) {
				magnets[i][j] = input.charAt(j) - '0';
			}
		}

		int K = Integer.parseInt(br.readLine());
		for (int i = 0; i < K; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			int idx = Integer.parseInt(st.nextToken()) - 1;
			int dir = Integer.parseInt(st.nextToken());

			setDirection(idx, dir);
			rotate();
		}

		int answer = getScore(magnets);
		System.out.println(answer);
	}

	// íšŒì „ ë°©í–¥ ì„¤ì •
	private static void setDirection(int idx, int dir) {
		Arrays.fill(dirs, 0);
		dirs[idx] = dir;

		// í˜„ì¬ ìì„ ê¸°ì¤€ ì˜¤ë¥¸ìª½ ì¡°íšŒ
		for (int i = idx + 1; i < 4; i++) {
			if (magnets[i][6] != magnets[i - 1][2]) {
				dirs[i] = dirs[i - 1] * -1;
			} else {
				break;
			}
		}

		// í˜„ì¬ ìì„ ê¸°ì¤€ ì™¼ìª½ ì¡°íšŒ
		for (int i = idx - 1; i >= 0; i--) {
			if (magnets[i][2] != magnets[i + 1][6]) {
				dirs[i] = dirs[i + 1] * -1;
			} else {
				break;
			}
		}
	}

	// íšŒì „
	private static void rotate() {
		for (int i = 0; i < 4; i++) {
			if (dirs[i] == 0) {
				continue;
			} else if (dirs[i] == 1) {
				int temp = magnets[i][7];
				for (int j = 7; j > 0; j--) {
					magnets[i][j] = magnets[i][j - 1];
				}
				magnets[i][0] = temp;
			} else if (dirs[i] == -1) {
				int temp = magnets[i][0];
				for (int j = 0; j < 7; j++) {
					magnets[i][j] = magnets[i][j + 1];
				}
				magnets[i][7] = temp;
			}
		}
	}

	private static int getScore(int[][] magnets) {
		int sum = 0;
		for (int i = 0; i < 4; i++) {
			if (magnets[i][0] == 1) {
				sum += Math.pow(2, i);
			}
		}

		return sum;
	}
}
