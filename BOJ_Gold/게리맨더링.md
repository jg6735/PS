import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class BJ_17471_ê²Œë¦¬ë§¨ë”ë§ {

	private static int N, answer;
	private static int[] population;
	private static boolean[] isSelected;
	private static int[][] adjMatrix;
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st;
		
		N = Integer.parseInt(br.readLine());
		population = new int[N + 1];
		isSelected = new boolean[N + 1];
		answer = Integer.MAX_VALUE;
		st = new StringTokenizer(br.readLine(), " ");
		for (int i = 1; i <= N; i++) {
			population[i] = Integer.parseInt(st.nextToken());
		}
		
		adjMatrix = new int[N + 1][N + 1];
		for (int i = 1; i <= N; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			int areaCount = Integer.parseInt(st.nextToken());
			for (int j = 0; j < areaCount; j++) {
				int areaNo = Integer.parseInt(st.nextToken());
				adjMatrix[i][areaNo] = 1;
			}
		}
		
		subset(1);
		
		if (answer == Integer.MAX_VALUE) {
			System.out.println(-1);
		} else {
			System.out.println(answer);
		}
	}

	private static void subset(int cnt) {
		if (cnt == N + 1) {
			if (check(true) && check(false)) {
				int arr1 = 0;
				int arr2 = 0;
				
				for (int i = 1; i <= N; i++) {
					if (isSelected[i]) {
						arr1 += population[i];
					} else {
						arr2 += population[i];
					}
				}
				
				int result = Math.abs(arr1 - arr2);
				answer = Math.min(result, answer);
			}
			return;
		}
		
		// í¬í•¨ / ë¹„í¬í•¨
		isSelected[cnt] = true;
		subset(cnt + 1);
		isSelected[cnt] = false;
		subset(cnt + 1);
	}
	
	private static boolean check(boolean flag) {
		boolean[] visited = new boolean[N + 1];
		Queue<Integer> queue = new LinkedList<Integer>();
		
		for (int i = 1; i <= N; i++) {
			if (isSelected[i] == flag) {
				queue.offer(i);
				visited[i] = true;
				break;
			}
		}
		
		while (!queue.isEmpty()) {
			int cur = queue.poll();
			
			for (int i = 1; i <= N; i++) {
				if (visited[i]) {
					continue;
				}
				
				if (flag != isSelected[i]) {
					continue;
				}
				
				if (adjMatrix[cur][i] != 1) {
					continue;
				}
				
				queue.offer(i);
				visited[i] = true;
			}
			
		}
		
		for (int i = 1; i <= N; i++) {
			if (!visited[i] && flag == isSelected[i]) {
				return false;
			}
		}			
		
		return true;
	}
}
