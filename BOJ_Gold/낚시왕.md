            int s = Integer.parseInt(st.nextToken());

        // ê°€ê¹Œìš´ìª½ë¶€í„° í–‰ íƒìƒ‰

            // ìƒì–´ê°€ ìˆë‹¤ë©´ ìƒì–´ í¬ê¸° ëˆ„ì í•˜ê³  í•´ë‹¹ ìƒì–´ ê°ì²´ ì§€ìš°ê¸°

    private static void move() {
